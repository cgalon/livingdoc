Exemple de diagramme d'état

	
	stateDiagram
	[*] --> First
	First --> Second
	First --> Third

	state First {
	[*] --> fir
	fir --> [*]
	}
	state Second {
	[*] --> sec
	sec --> [*]
	}
	state Third {
	[*] --> thi
	thi --> [*]
	}
stateDiagram [*] --> First First --> Second First --> Third state First { [*] --> fir fir --> [*] } state Second { [*] --> sec sec --> [*] } state Third { [*] --> thi thi --> [*] }
