    anychart.onDocumentLoad(function () {
        // create an instance of a pie chart
        var chart = anychart.pie();
        // set the data
        chart.data([
            ["Organisation", 5],
            ["Initialisation", 3],
            ["Environnement", 6],
            ["Test", 1],
            ["Conteneurisation", 5],
            ["Déploiement", 9]
        ]);
        // set chart title
        chart.title("Les livrables TMJ-M par thème");
        // set the container element
        chart.container("exemple-simple");
        // initiate chart display
        chart.draw();
    });
